# SP6: End-to-End Test Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire `make e2e` to run both PA entry paths (CRD→DTR→PAS via Playwright + direct PAS via pytest) against the real deployed stack, with full lifecycle assertions through auto-approval and `$inquire` ClaimResponse.

**Architecture:** Two test suites orchestrated by a single `make e2e` target. A seed script writes a fixture manifest (JSON) consumed by both suites. The Keycloak issuer mismatch (browser tokens vs. interop validator) is resolved by pinning `KC_HOSTNAME_URL` and aligning `KEYCLOAK_ISSUER_URI`. The `$inquire` endpoint (`PasClaimInquireProvider`) is the final ClaimResponse assertion surface — no HAPI write-back needed.

**Tech Stack:** Python 3.12 · httpx · pytest-asyncio · pytest-json-report · Playwright (TypeScript) · Spring Boot HAPI FHIR (interop) · Keycloak 24 · Docker Compose

## Key facts (read before implementing)

- **BFF route prefix:** `main.py` mounts cases with `prefix="/bff"` → full path is `http://localhost:8001/bff/cases/{case_id}`. Nginx keeps the `/bff/` prefix when proxying.
- **ClaimResponse assertion:** use `GET /fhir/Claim/{correlation_id}/$inquire` (not a FHIR search). `PasClaimInquireProvider` reads `DecisionRecord.outcome` — null = QUEUED, non-null = COMPLETE.
- **DecisionRecord update:** `DecisionEventConsumer` (Spring `@KafkaListener`) subscribes to `decision.recorded` and sets `outcome` when the workflow engine auto-approves. The $inquire call is only ready after this consumer fires.
- **case_id from $submit:** the `ClaimResponse` in the $submit response Bundle carries extension `https://enstellar.simintero.com/fhir/StructureDefinition/case-id` with the workflow case UUID.
- **correlation_id / bundle ID:** the bundle `id` field passed to $submit is used as `correlationId` in `$inquire`. Set it deterministically to `e2e-bundle-001`.
- **Mock Digicore** always returns `decision=approved` → all E2E cases auto-approve.
- **BFF iss validation:** python-jose `jwt.decode` does NOT validate `iss` by default — no BFF code change needed for the issuer fix.
- **Keycloak client:** reuse existing `enstellar-test-client` in CI overlay (`directAccessGrantsEnabled: true`, `serviceAccountsEnabled: true`). Use `password` grant for all E2E calls (seed script, pytest, Playwright) with the `e2e-reviewer` user.
- **Keycloak roles claim:** BFF checks `payload.get("roles", [])` (top-level). Keycloak by default puts realm roles in `realm_access.roles`. A `roles_flat_mapper` must be added to the `fhir_scopes` client scope in the realm JSON to map realm roles to a flat `roles` claim.
- **tenant_id claim:** already mapped via `tenant_id_mapper` in `fhir_scopes` scope from user attribute.

```python
#!/usr/bin/env python3
"""E2E fixture seed. Run with: uv run --project services/workflow-engine python scripts/e2e_seed.py"""
from __future__ import annotations

import json
import os
import sys
import time
import urllib.request
import urllib.parse

INTEROP_BASE = "http://localhost:8080"
KEYCLOAK_URL = "http://localhost:8081/realms/enstellar/protocol/openid-connect/token"
MANIFEST_PATH = "infra/compose/e2e-fixture-manifest.json"

PATIENT_ID      = "e2e-patient-001"
PRACTITIONER_ID = "e2e-practitioner-001"
ORG_ID          = "e2e-org-001"
COVERAGE_ID     = "e2e-coverage-001"
CLAIM_ID        = "e2e-claim-001"
BUNDLE_ID       = "e2e-bundle-001"   # used as correlationId for $inquire


def get_token() -> str:
    data = urllib.parse.urlencode({
        "grant_type": "password",
        "client_id": "enstellar-test-client",
        "client_secret": os.environ.get("TEST_CLIENT_SECRET", "test-secret"),
        "username": "e2e-reviewer",
        "password": os.environ["E2E_PASSWORD"],
    }).encode()
    req = urllib.request.Request(KEYCLOAK_URL, data=data)
    with urllib.request.urlopen(req, timeout=15) as r:
        return json.loads(r.read())["access_token"]


def fhir_put(token: str, resource_type: str, resource_id: str, body: dict) -> None:
    url = f"{INTEROP_BASE}/fhir/{resource_type}/{resource_id}"
    data = json.dumps(body).encode()
    req = urllib.request.Request(url, data=data, method="PUT")
    req.add_header("Authorization", f"Bearer {token}")
    req.add_header("Content-Type", "application/fhir+json")
    req.add_header("Accept", "application/fhir+json")
    req.add_header("X-Tenant-Id", "tenant-dev")
    with urllib.request.urlopen(req, timeout=30) as r:
        status = r.status
    print(f"  PUT {resource_type}/{resource_id} → {status}")


def pas_submit(token: str) -> dict:
    bundle = {
        "resourceType": "Bundle",
        "id": BUNDLE_ID,
        "type": "collection",
        "entry": [
            {"resource": {
                "resourceType": "Claim",
                "id": CLAIM_ID,
                "use": "preauthorization",
                "status": "active",
                "patient": {"reference": f"Patient/{PATIENT_ID}"},
                "provider": {"reference": f"Practitioner/{PRACTITIONER_ID}"},
                "insurer": {"reference": f"Organization/{ORG_ID}"},
                "insurance": [{"sequence": 1, "focal": True, "coverage": {"reference": f"Coverage/{COVERAGE_ID}"}}],
                "diagnosis": [{"sequence": 1, "diagnosisCodeableConcept": {"coding": [
                    {"system": "http://hl7.org/fhir/sid/icd-10-cm", "code": "M54.5", "display": "Low back pain"}
                ]}}],
                "item": [{"sequence": 1,
                    "productOrService": {"coding": [
                        {"system": "http://www.ama-assn.org/go/cpt", "code": "99213", "display": "Office visit"}
                    ]},
                    "diagnosisSequence": [1]}],
            }},
            {"resource": {"resourceType": "Patient", "id": PATIENT_ID,
                "name": [{"family": "E2ESmith", "given": ["Jane"]}]}},
            {"resource": {"resourceType": "Practitioner", "id": PRACTITIONER_ID,
                "name": [{"family": "E2EJones", "given": ["Bob"]}]}},
            {"resource": {"resourceType": "Organization", "id": ORG_ID,
                "name": "E2E Health Plan"}},
            {"resource": {"resourceType": "Coverage", "id": COVERAGE_ID,
                "status": "active",
                "subscriber": {"reference": f"Patient/{PATIENT_ID}"},
                "beneficiary": {"reference": f"Patient/{PATIENT_ID}"},
                "payor": [{"reference": f"Organization/{ORG_ID}"}]}},
        ],
    }
    url = f"{INTEROP_BASE}/fhir/Claim/$submit"
    data = json.dumps(bundle).encode()
    req = urllib.request.Request(url, data=data, method="POST")
    req.add_header("Authorization", f"Bearer {token}")
    req.add_header("Content-Type", "application/fhir+json")
    req.add_header("Accept", "application/fhir+json")
    req.add_header("X-Tenant-Id", "tenant-dev")
    with urllib.request.urlopen(req, timeout=30) as r:
        return json.loads(r.read())


def extract_case_id(submit_response: dict) -> str:
    """Extract case_id from the $submit ClaimResponse extension."""
    for entry in submit_response.get("entry", []):
        resource = entry.get("resource", {})
        if resource.get("resourceType") == "ClaimResponse":
            for ext in resource.get("extension", []):
                if ext.get("url") == "https://enstellar.simintero.com/fhir/StructureDefinition/case-id":
                    return ext["valueString"]
    raise ValueError("case-id extension not found in $submit response")


def main() -> None:
    print("→ Acquiring token...")
    token = get_token()

    print("→ Seeding FHIR resources...")
    fhir_put(token, "Patient", PATIENT_ID, {
        "resourceType": "Patient", "id": PATIENT_ID,
        "name": [{"family": "E2ESmith", "given": ["Jane"]}],
        "gender": "female", "birthDate": "1980-01-15",
    })
    fhir_put(token, "Practitioner", PRACTITIONER_ID, {
        "resourceType": "Practitioner", "id": PRACTITIONER_ID,
        "name": [{"family": "E2EJones", "given": ["Bob"]}],
    })
    fhir_put(token, "Organization", ORG_ID, {
        "resourceType": "Organization", "id": ORG_ID, "name": "E2E Health Plan",
    })
    fhir_put(token, "Coverage", COVERAGE_ID, {
        "resourceType": "Coverage", "id": COVERAGE_ID, "status": "active",
        "subscriber": {"reference": f"Patient/{PATIENT_ID}"},
        "beneficiary": {"reference": f"Patient/{PATIENT_ID}"},
        "payor": [{"reference": f"Organization/{ORG_ID}"}],
    })

    print("→ Calling PAS $submit...")
    # If duplicate (422), the DecisionRecord already exists from a prior seed run.
    try:
        submit_resp = pas_submit(token)
    except urllib.error.HTTPError as e:
        if e.code == 422:
            print("  $submit 422 — duplicate bundle (prior seed run). Using known IDs.")
            case_id = None  # Will be resolved via $inquire after tests run
        else:
            raise
    else:
        case_id = extract_case_id(submit_resp)
        print(f"  case_id = {case_id}")

    manifest = {
        "patient_id":      PATIENT_ID,
        "practitioner_id": PRACTITIONER_ID,
        "org_id":          ORG_ID,
        "coverage_id":     COVERAGE_ID,
        "case_id":         case_id,
        "claim_id":        CLAIM_ID,
        "correlation_id":  BUNDLE_ID,
    }
    with open(MANIFEST_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"✓ Manifest written to {MANIFEST_PATH}")


if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""E2E fixture teardown. Safe to run after `make down`; ignores 404."""
from __future__ import annotations

import json
import os
import urllib.request
import urllib.parse

INTEROP_BASE = "http://localhost:8080"
KEYCLOAK_URL = "http://localhost:8081/realms/enstellar/protocol/openid-connect/token"

RESOURCES = [
    ("Coverage",     "e2e-coverage-001"),
    ("Organization", "e2e-org-001"),
    ("Practitioner", "e2e-practitioner-001"),
    ("Patient",      "e2e-patient-001"),
]


def get_token() -> str:
    data = urllib.parse.urlencode({
        "grant_type": "password",
        "client_id": "enstellar-test-client",
        "client_secret": os.environ.get("TEST_CLIENT_SECRET", "test-secret"),
        "username": "e2e-reviewer",
        "password": os.environ["E2E_PASSWORD"],
    }).encode()
    req = urllib.request.Request(KEYCLOAK_URL, data=data)
    with urllib.request.urlopen(req, timeout=15) as r:
        return json.loads(r.read())["access_token"]


def fhir_delete(token: str, resource_type: str, resource_id: str) -> None:
    url = f"{INTEROP_BASE}/fhir/{resource_type}/{resource_id}"
    req = urllib.request.Request(url, method="DELETE")
    req.add_header("Authorization", f"Bearer {token}")
    req.add_header("X-Tenant-Id", "tenant-dev")
    try:
        with urllib.request.urlopen(req, timeout=15) as r:
            print(f"  DELETE {resource_type}/{resource_id} → {r.status}")
    except urllib.error.HTTPError as e:
        if e.code == 404:
            print(f"  DELETE {resource_type}/{resource_id} → 404 (already gone)")
        else:
            print(f"  DELETE {resource_type}/{resource_id} → {e.code} (ignored)")


def main() -> None:
    print("→ Acquiring token...")
    try:
        token = get_token()
    except Exception as e:
        print(f"  Token acquisition failed ({e}) — stack may be down; skipping teardown.")
        return

    print("→ Deleting E2E FHIR fixtures...")
    for resource_type, resource_id in RESOURCES:
        fhir_delete(token, resource_type, resource_id)
    print("✓ Teardown complete")


if __name__ == "__main__":
    main()
```

```python
from __future__ import annotations

import asyncio
import json
import os
from pathlib import Path

import httpx

KEYCLOAK_TOKEN_URL = "http://localhost:8081/realms/enstellar/protocol/openid-connect/token"
BFF_BASE = "http://localhost:8001"
INTEROP_BASE = "http://localhost:8080"
TENANT_HEADER = {"X-Tenant-Id": "tenant-dev"}


def load_manifest(path: str | None = None) -> dict:
    manifest_path = path or os.environ.get(
        "E2E_MANIFEST", "infra/compose/e2e-fixture-manifest.json"
    )
    return json.loads(Path(manifest_path).read_text())


async def get_token(
    username: str | None = None,
    password: str | None = None,
) -> str:
    async with httpx.AsyncClient(timeout=15) as client:
        r = await client.post(
            KEYCLOAK_TOKEN_URL,
            data={
                "grant_type": "password",
                "client_id": "enstellar-test-client",
                "client_secret": os.environ.get("TEST_CLIENT_SECRET", "test-secret"),
                "username": username or os.environ.get("E2E_USER", "e2e-reviewer"),
                "password": password or os.environ["E2E_PASSWORD"],
            },
        )
        r.raise_for_status()
        return r.json()["access_token"]


async def wait_for_case_state(
    case_id: str,
    token: str,
    target: str = "approved",
    timeout: int = 60,
    interval: int = 2,
) -> dict:
    """Poll BFF GET /bff/cases/{case_id} until status == target.

    Raises TimeoutError with a descriptive message if not reached within timeout.
    """
    headers = {"Authorization": f"Bearer {token}", **TENANT_HEADER}
    deadline = asyncio.get_event_loop().time() + timeout
    async with httpx.AsyncClient(timeout=10) as client:
        while True:
            r = await client.get(f"{BFF_BASE}/bff/cases/{case_id}", headers=headers)
            if r.status_code == 200:
                data = r.json()
                status = data.get("status", "")
                if status == target:
                    return data
            remaining = deadline - asyncio.get_event_loop().time()
            if remaining <= 0:
                last_status = r.json().get("status", "unknown") if r.status_code == 200 else r.status_code
                raise TimeoutError(
                    f"Case {case_id} did not reach '{target}' within {timeout}s "
                    f"(last observed: {last_status})"
                )
            await asyncio.sleep(interval)
```

```python
"""Path B E2E test: direct PAS $submit → workflow auto-approve → $inquire ClaimResponse."""
from __future__ import annotations

import pytest
import httpx

from tests.e2e.helpers import INTEROP_BASE, TENANT_HEADER, wait_for_case_state


@pytest.mark.asyncio
async def test_direct_pas_full_lifecycle(manifest: dict, token: str) -> None:
    headers = {"Authorization": f"Bearer {token}", **TENANT_HEADER,
               "Content-Type": "application/fhir+json", "Accept": "application/fhir+json"}

    case_id     = manifest["case_id"]
    correlation_id = manifest["correlation_id"]   # e2e-bundle-001

    # ── 1. Seed already ran $submit; case_id is in the manifest. ──────────────
    assert case_id, "case_id missing from manifest — run make e2e-seed first"

    # ── 2. Poll workflow until case reaches 'approved' ─────────────────────────
    case = await wait_for_case_state(case_id, token, target="approved", timeout=60)
    assert case["status"] == "approved", f"Expected approved, got {case['status']}"

    # ── 3. Assert decision recorded ───────────────────────────────────────────
    events = case.get("events", [])
    decision_events = [e for e in events if e.get("event_type") == "decision_recorded" or
                       e.get("type") == "decision_recorded"]
    assert decision_events, f"No decision_recorded event found in case events: {events}"
    decision_event = decision_events[0]
    payload = decision_event.get("payload", {})
    assert payload.get("outcome") == "approved", f"Unexpected decision outcome: {payload}"
    assert payload.get("auto_approved") is True, "Expected auto_approved=True"

    # ── 4. Assert comms fired (notification event present) ────────────────────
    comms_events = [e for e in events if "notif" in str(e.get("event_type", "")).lower() or
                    "notif" in str(e.get("type", "")).lower()]
    assert comms_events, f"No notification event found in case events: {events}"

    # ── 5. Assert $inquire returns complete ClaimResponse ─────────────────────
    async with httpx.AsyncClient(timeout=15) as client:
        r = await client.get(
            f"{INTEROP_BASE}/fhir/Claim/{correlation_id}/$inquire",
            headers={**headers},
        )
    assert r.status_code == 200, f"$inquire returned {r.status_code}: {r.text[:200]}"

    inquire_bundle = r.json()
    assert inquire_bundle["resourceType"] == "Bundle", "Expected Bundle from $inquire"

    claim_responses = [
        e["resource"] for e in inquire_bundle.get("entry", [])
        if e.get("resource", {}).get("resourceType") == "ClaimResponse"
    ]
    assert claim_responses, "No ClaimResponse in $inquire bundle"
    cr = claim_responses[0]
    assert cr["outcome"] == "complete", f"Expected outcome=complete, got {cr.get('outcome')}"
```

### Background
The Playwright test authenticates with Keycloak via a resource-owner password grant (no browser login UI). It drives the `/ehr-sim` page to fire a CDS Hook, clicks the DTR link, fills the questionnaire, then polls the BFF for case approval and checks `$inquire`.

The seeded Patient ID is `e2e-patient-001`. The mock Digicore questionnaire for service code `99213` has three required fields: `indication` (string), `tried-conservative` (boolean), `diagnosis` (string).

- [ ] **Step 1: Create `apps/web/e2e/real-stack/ehr-to-pa.spec.ts`**

```typescript
import { test, expect, request as playwrightRequest } from '@playwright/test'
import fs from 'fs'
import path from 'path'

// ── Manifest and auth ──────────────────────────────────────────────────────────

function loadManifest(): Record<string, string> {
  const manifestPath = process.env.E2E_MANIFEST ??
    path.resolve(__dirname, '../../../../infra/compose/e2e-fixture-manifest.json')
  return JSON.parse(fs.readFileSync(manifestPath, 'utf-8'))
}

async function getToken(): Promise<string> {
  const ctx = await playwrightRequest.newContext()
  const resp = await ctx.post(
    'http://localhost:8081/realms/enstellar/protocol/openid-connect/token',
    {
      form: {
        grant_type: 'password',
        client_id: 'enstellar-test-client',
        client_secret: process.env.TEST_CLIENT_SECRET ?? 'test-secret',
        username: process.env.E2E_USER ?? 'e2e-reviewer',
        password: process.env.E2E_PASSWORD ?? 'e2e-pass',
      },
    }
  )
  const body = await resp.json()
  await ctx.dispose()
  return body.access_token as string
}

// ── Poll helper ────────────────────────────────────────────────────────────────

async function waitForCaseState(
  caseId: string,
  token: string,
  target: string,
  timeoutMs = 60_000,
): Promise<Record<string, unknown>> {
  const ctx = await playwrightRequest.newContext()
  const deadline = Date.now() + timeoutMs
  try {
    while (Date.now() < deadline) {
      const resp = await ctx.get(`http://localhost:8001/bff/cases/${caseId}`, {
        headers: { Authorization: `Bearer ${token}`, 'X-Tenant-Id': 'tenant-dev' },
      })
      if (resp.ok()) {
        const data = await resp.json() as Record<string, unknown>
        if (data['status'] === target) return data
      }
      await new Promise(r => setTimeout(r, 2000))
    }
    throw new Error(`Case ${caseId} did not reach '${target}' within ${timeoutMs}ms`)
  } finally {
    await ctx.dispose()
  }
}

// ── Tests ──────────────────────────────────────────────────────────────────────

let manifest: Record<string, string>
let token: string

test.beforeAll(async () => {
  manifest = loadManifest()
  token = await getToken()
})

test('CRD → DTR → PAS full lifecycle', async ({ page }) => {
  // Inject auth token for all BFF calls made by the browser
  await page.setExtraHTTPHeaders({ Authorization: `Bearer ${token}` })

  // ── 1. Fire CDS Hook via EHR sim ────────────────────────────────────────────
  await page.goto('/ehr-sim')
  await page.fill('[data-testid="patient-id-input"], input[placeholder*="patient"], input[name*="patient"]',
    manifest['patient_id'])
  await page.fill('[data-testid="service-code-input"], input[placeholder*="service"], input[name*="service"]',
    '99213')
  await page.click('button:has-text("Check Coverage")')

  // ── 2. Assert CRD card appears ──────────────────────────────────────────────
  const cardLocator = page.locator('[data-testid="crd-card"], .crd-card, [class*="card"]').first()
  await expect(cardLocator).toBeVisible({ timeout: 15_000 })
  const cardText = await cardLocator.textContent()
  expect(cardText?.toLowerCase()).toContain('prior auth')

  // ── 3. Launch DTR ────────────────────────────────────────────────────────────
  const dtrLink = page.locator('a:has-text("DTR"), a:has-text("documentation"), button:has-text("DTR")').first()
  await dtrLink.click()
  await page.waitForURL('**/dtr-form**', { timeout: 10_000 })
  const questionLocator = page.locator('[class*="question"], form label, .questionnaire').first()
  await expect(questionLocator).toBeVisible({ timeout: 10_000 })

  // ── 4. Fill and submit questionnaire ─────────────────────────────────────────
  // indication (string)
  const indicationInput = page.locator('input[type="text"], textarea').first()
  await indicationInput.fill('Chronic low back pain requiring therapeutic exercise')

  // tried-conservative (boolean — checkbox or toggle)
  const conservativeCheck = page.locator('input[type="checkbox"]').first()
  await conservativeCheck.check()

  // diagnosis (string — second text input)
  const diagnosisInput = page.locator('input[type="text"], textarea').nth(1)
  await diagnosisInput.fill('M54.5')

  await page.click('button:has-text("Submit"), button[type="submit"]')

  // Assert success (no error banner)
  await expect(page.locator('[class*="error"], [role="alert"][class*="error"]')).toHaveCount(0)
  await expect(page.locator('text=/submit|complet|success/i')).toBeVisible({ timeout: 10_000 })

  // ── 5. Poll workflow for approved ────────────────────────────────────────────
  const caseId = manifest['case_id']
  expect(caseId).toBeTruthy()
  const caseData = await waitForCaseState(caseId, token, 'approved', 60_000)
  expect(caseData['status']).toBe('approved')

  // ── 6. Assert $inquire returns complete ClaimResponse ────────────────────────
  const correlationId = manifest['correlation_id']
  const ctx = await playwrightRequest.newContext()
  const inquireResp = await ctx.get(
    `http://localhost:8080/fhir/Claim/${correlationId}/$inquire`,
    {
      headers: {
        Authorization: `Bearer ${token}`,
        Accept: 'application/fhir+json',
        'X-Tenant-Id': 'tenant-dev',
      },
    }
  )
  await ctx.dispose()
  expect(inquireResp.ok()).toBe(true)
  const inquireBundle = await inquireResp.json() as Record<string, unknown>
  const entries = (inquireBundle['entry'] as unknown[]) ?? []
  const claimResponse = entries
    .map(e => (e as Record<string, unknown>)['resource'] as Record<string, unknown>)
    .find(r => r?.['resourceType'] === 'ClaimResponse')
  expect(claimResponse).toBeTruthy()
  expect(claimResponse!['outcome']).toBe('complete')
})
```
